    mes "...................";

      set lastClass$,class;
      jobchange getarg(0);

      jobchange getarg(0);
    }
    specialeffect 435;
    callfunc "SRO_JobchangerEnd";

  u_notsure:
    mes "[เทพี วาลคิรี]";
    mes "...................";
    callfunc "SRO_JobchangerEnd";
}
